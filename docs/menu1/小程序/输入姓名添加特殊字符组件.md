## 组件封装

### index.wxml
```html
<view class="special" catch:tap="addSpecial">·</view>
```

### index.js
```javascript
Component({
  methods: {
    addSpecial() {
      this.triggerEvent('addSpecial', '·')
    }
  }
})
```

## 调用

### index.wxml
```html
<van-field 
  value="{{form.fullname}}" 
  hold-keyboard 
  input-align="right" 
  label="真实姓名" 
  placeholder="请输入真实姓名"
  maxlength="{{20}}"
  required
>
  <special-name slot="button" bind:addSpecial="addSpecial"></special-name>
</van-field>
```

### index.js
```javascript
  data: {
    form: {
      fullname: ''
    }
  }
  // 姓名添加特殊字符
  addSpecial(e) {
    this.setData({
      'form.fullname': this.data.form.fullname + '·'
    })
  },
```

```json
{
  "navigationStyle": "custom",
  "usingComponents": {
    "special-name": "../../components/specialName/index",
  },
}
```
